using System;
using System.Media;
using System.Windows.Forms;
using System.Diagnostics;

namespace MouseMoveErrorPopup
{
    static class Program
    {
        [STAThread]
        static void Main()
        {
            Application.EnableVisualStyles();
            Application.SetCompatibleTextRenderingDefault(false);
            Application.Run(new MainForm());
        }
    }

    public class MainForm : Form
    {
        // Path to the Windows XP error sound (must exist on disk)
        private readonly string xpErrorSoundPath = @"C:\Path\To\WindowsXpError.wav";
        private readonly TimeSpan debounce = TimeSpan.FromMilliseconds(700);
        private DateTime lastShown = DateTime.MinValue;

        public MainForm()
        {
            Text = "MouseMove -> XP error popup";
            Width = 600;
            Height = 400;
            this.MouseMove += MainForm_MouseMove;
        }

        private void MainForm_MouseMove(object sender, MouseEventArgs e)
        {
            var now = DateTime.UtcNow;
            if (now - lastShown < debounce) return; // simple debounce

            lastShown = now;

            try
            {
                // Play the WAV synchronously so the sound starts before the dialog.
                using (var player = new SoundPlayer(xpErrorSoundPath))
                {
                    player.PlaySync(); // blocks until sound finishes; use Play() for async
                }
            }
            catch (Exception ex)
            {
                // If the WAV can't be played, fallback to the system Hand sound
                Debug.WriteLine("Failed to play WAV: " + ex);
                System.Media.SystemSounds.Hand.Play();
            }

            // Show an error MessageBox (error icon)
            MessageBox.Show(this, "An error occurred.", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
        }
    }
}
